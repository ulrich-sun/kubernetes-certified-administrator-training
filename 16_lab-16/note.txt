##################### upgrade from 1.29.1 to 1.30.1 ############################
# procedure reference https://kubernetes.io/docs/tasks/administer-cluster/kubeadm/kubeadm-upgrade/
# https://kubernetes.io/docs/reference/using-api/deprecation-policy/
# https://github.com/kubernetes/kubernetes/tree/master/CHANGELOG
### IMPORTANT : Snaphost VMs before start

kubeadm version
kubeadm upgrade plan
yum list --showduplicates kubeadm --disableexcludes=kubernetes

cat <<EOF | sudo tee /etc/yum.repos.d/kubernetes.repo
[kubernetes]
name=Kubernetes
baseurl=https://pkgs.k8s.io/core:/stable:/v1.30/rpm/
enabled=1
gpgcheck=1
gpgkey=https://pkgs.k8s.io/core:/stable:/v1.30/rpm/repodata/repomd.xml.key
exclude=kubelet kubeadm kubectl cri-tools kubernetes-cni
EOF

# clean yum cache
rm -rf /var/cache/yum/*
yum clean metadata
yum clean all

yum update
yum list --showduplicates kubeadm --disableexcludes=kubernetes
sudo yum install -y  kubeadm-1.30.1 --disableexcludes=kubernetes
kubeadm version
kubeadm upgrade plan

kubeadm upgrade apply v1.30.1

kubectl get nodes

kubectl drain master --ignore-daemonsets
yum install -y kubelet-1.30.1 kubectl-1.30.1 --disableexcludes=kubernetes

sudo systemctl daemon-reload
sudo systemctl restart kubelet

kubectl uncordon master

kubectl get nodes

# on worker nodes

kubeadm version
yum list --showduplicates kubeadm --disableexcludes=kubernetes

cat <<EOF | sudo tee /etc/yum.repos.d/kubernetes.repo
[kubernetes]
name=Kubernetes
baseurl=https://pkgs.k8s.io/core:/stable:/v1.30/rpm/
enabled=1
gpgcheck=1
gpgkey=https://pkgs.k8s.io/core:/stable:/v1.30/rpm/repodata/repomd.xml.key
exclude=kubelet kubeadm kubectl cri-tools kubernetes-cni
EOF

# clean yum cache
rm -rf /var/cache/yum/*
yum clean metadata
yum clean all

yum update
sudo yum install -y  kubeadm-1.30.1 --disableexcludes=kubernetes

kubeadm upgrade node

# on master node
kubectl drain worker1 --ignore-daemonsets

# on worker nodes
yum install -y kubelet-1.30.1 --disableexcludes=kubernetes

sudo systemctl daemon-reload
sudo systemctl restart kubelet

# on master node
kubectl uncordon worker1

kubectl get nodes
